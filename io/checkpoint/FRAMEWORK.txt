/**
@defgroup chkpoint  Checkpointing routines
@ingroup io

@brief Checkopinting routines for toolbox

@details This is the main interface to I/O checkpinting.

@mod_interface
@interface_list Interface provided:
-# @ref chkpt_register
-# @ref chkpt_init
-# @ref chkpt_main

@par
@interface_reqr Interface required:
-# chkpts_register
-# chkpts_init
-# chkpts_read
-# chkpts_write

@mod_interface_example
@code{.f}
      subroutine userchk
      include 'TSTEP'
      
      if (ISTEP.eq.0) then
!     register modules and parameters
         call mntr_register_mod
         call rprm_register
         call mntr_register_par
         call chkpt_register
         call rprm_dict_get
         call mntr_init
         call chkpt_init
      endif

!     save/load files for full-restart
      call chkpt_main

      return
      end
@endcode

@mod_param
@param_list Parameters provided by checkpoint module (include file - \a CHKPOINTD):
|Varaible | Type | Runtime parameter | Description |
|---------|------|-------------------|-------------|
| chpt_ifrst | logical | _chkpoint:readchkpt | start simulation form checkpoint |
| chpt_fnum | integer | _chkpoint:chkpFnumber | restart file number |
| chpt_step | integer | _chkpoint:chkpInterval | checkpiont dump frequency (number of time steps) |
| chpt_wtimes | string | _chkpoint:WallTime | simulation wall time |


@mod_param_example
@code{.txt}
[_CHKPOINT]
chkpInterval = 10
readchkpt = no
chkpFnumber = 1
WallTime = 02:05
@endcode

*/
